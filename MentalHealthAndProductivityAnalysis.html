# Step 4: Finding out if Treatment Help Reduce Work Interference?

# Comparing work interference between those who received treatment and those who didn’t
# Group data by treatment status and work interference
import matplotlib.pyplot as plt
import numpy as np

# Set dark mode
plt.style.use("dark_background")

# Create figure with dark blue background
fig, ax = plt.subplots(figsize=(8,5))
fig.patch.set_facecolor("darkblue")  # Figure background
ax.set_facecolor("darkblue")  # Plot background

# Group data by treatment status and work interference
treatment_vs_interference = df.groupby("treatment")["work_interfere"].value_counts().unstack()

# Generate shades of blue for bars
num_bars = len(treatment_vs_interference.columns)
shades_of_blue = plt.cm.Blues(np.linspace(0.4, 1, num_bars))  # Lighter to darker blue

# Plot stacked bar chart
treatment_vs_interference.plot(kind="bar", figsize=(8,5), color=shades_of_blue, edgecolor="white", ax=ax)

# Labels and title
ax.set_xlabel("Taking Mental Health Treatment", color="white")
ax.set_ylabel("Number of People", color="white")
ax.set_title("Impact of Mental Health Treatment on Work Interference", color="white")
ax.tick_params(colors="white")  # White tick labels
ax.legend(title="Work Interference Level", labelcolor="white", facecolor="darkblue", edgecolor="white")  
plt.xticks(rotation=0)
plt.grid(axis="y", linestyle="--", alpha=0.7, color="white")  # Light white grid
plt.show()

#  Step 5: Find out does Workplace Support Improve Mental Health?

# Check how workplace benefits affect mental health treatment
import matplotlib.pyplot as plt
import numpy as np

# Set dark mode
plt.style.use("dark_background")

# Create figure with dark blue background
fig, ax = plt.subplots(figsize=(8,5))
fig.patch.set_facecolor("darkblue")  # Set figure background
ax.set_facecolor("darkblue")  # Set plot background

benefits_treatment = df.groupby("benefits")["treatment"].value_counts().unstack()

# Generate shades of blue for bars
num_bars = len(benefits_treatment.columns)

# Step 6: Finding out what Factors Affect Work Interference the Most?

# Let us find out which factors impact work interference
# Convert categorical columns to numerical for correlation analysis
df_encoded = df.copy()
df_encoded["work_interfere"] = df_encoded["work_interfere"].astype("category").cat.codes
df_encoded["treatment"] = df_encoded["treatment"].astype("category").cat.codes
df_encoded["benefits"] = df_encoded["benefits"].astype("category").cat.codes
df_encoded["care_options"] = df_encoded["care_options"].astype("category").cat.codes

# Find correlation of work interference with other factors
correlation = df_encoded.corr(numeric_only=True)["work_interfere"].sort_values(ascending=False)
